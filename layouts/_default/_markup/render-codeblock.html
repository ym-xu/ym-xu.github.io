The marker is stripped from the displayed code. */ -}} {{- if strings.HasPrefix (strings.TrimLeft "\n" $code) "#| norun" -}} {{- $norun = true -}} {{- $code = strings.TrimLeft "\n" (strings.TrimPrefix "#| norun" (strings.TrimLeft "\n" $code)) -}} {{- end -}} {{- $runnable := and (in (slice "r" "R") $lang) (not $norun) -}}

{{ end }}
